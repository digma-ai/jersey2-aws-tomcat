
install latest maven

mvn clean package

deploy target/helloworld-webapp.war on tomcat 9


Docker:

mvn clean package

docker build -t digmaai/tomcat-webapp-jersey-aws .

docker run -p 8088:8080 digmaai/tomcat-webapp-jersey-aws

docker tag digmaai/tomcat-webapp-jersey-aws:latest digmaai/tomcat-webapp-jersey-aws:4.0

Push:
docker push digmaai/tomcat-webapp-jersey-aws:4.0
try it
docker run -p 8088:8080 digmaai/tomcat-webapp-jersey-aws:4.0
